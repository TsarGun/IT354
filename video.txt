//rest parameter
function fun1(a, b){ 
    return a + b; 
} 
console.log(fun1(1, 2)); // 3 
console.log(fun1(1, 2, 3, 4, 5)); 

// with rest parameter 
function fun2(...input){ 
    let sum = 0; 
    for(let i of input){ 
        sum+=i; 
    } 
    return sum; 
} 
console.log(fun2(1,2)); //3 
console.log(fun2(1,2,3)); //6 
console.log(fun2(1,2,3,4,5)); //15
